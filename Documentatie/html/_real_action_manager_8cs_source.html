87 public void SellProduct(int barcode, int quantity)
88 {
89 //verify if database contains a product with the given barcode
90 Product product = _dbInstance.SelectProduct(barcode);
91 if (product == null)
92 {
93 throw new RecordNotFoundException("This product doesn't exist.");
94 }
95 else
96 {
97 if (quantity <= 0)
98 {
99 throw new InvalidDataException("Quantity must be greater than zero.");
100 }
101 int newStock = product.Stock - quantity;
