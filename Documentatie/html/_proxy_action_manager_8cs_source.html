84 public bool Login(string username, string password)
85 {
86 List<User> users = _dbInstance.SelectAllUsers();
87 foreach (User user in users)
88 {
89 if (user.Username.Equals(username))
90 {
91 if (user.Password.Equals(Cryptography.HashString(password)))
92 {
93 _currentUser = user;
